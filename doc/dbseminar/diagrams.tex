% Invariant node.
\newcommand{\inode}[3][]{
  \node[draw, shape=circle, minimum width=0.6cm] (#2) at (#3) {};
  \node[fill, shape=circle, #1] () at (#3) {};
}

% Dashed node.
\newcommand{\dnode}[3][]{
  \node[draw, dashed, shape=circle, minimum width=0.6cm] (#2) at (#3) {};
  \node[fill, shape=circle, #1] () at (#3) {};
}

% Normal node.
\newcommand{\nnode}[3][]{
  \node[fill, shape=circle, #1] (#2) at (#3) {};
}
